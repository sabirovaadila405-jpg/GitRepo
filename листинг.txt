# task.py
from datetime import datetime
class Task:
    def __init__(self, title, completed=False, created_at=None):
        self.id = None  # будет присвоен при добавлении
        self.title = title
        self.completed = completed
        self.created_at = created_at or datetime.now().isoformat()

# storage.py
import json
def load_tasks(filename="tasks.json"):
    try:
        with open(filename, 'r', encoding='utf-8') as f:
            data = json.load(f)
            return [Task(**item) for item in data]
    except FileNotFoundError:
        return []

def save_tasks(tasks, filename="tasks.json"):
    with open(filename, 'w', encoding='utf-8') as f:
        json.dump([t.__dict__ for t in tasks], f, ensure_ascii=False, indent=2)

# main.py
def main():
    tasks = load_tasks()
    if tasks:
        max_id = max(t.id for t in tasks)
    else:
        max_id = 0
    while True:
        print("\n=== Менеджер задач ===")
        print("1. Показать все")
        print("2. Добавить")
        print("3. Отметить выполненной")
        print("4. Удалить")
        print("5. Фильтр")
        print("6. Выход")
        choice = input("Выберите действие: ")
        if choice == '1':
            for t in tasks:
                status = "✓" if t.completed else "✗"
                print(f"{t.id}: {t.title} [{status}] {t.created_at}")
        elif choice == '2':
            title = input("Введите название задачи: ")
            max_id += 1
            new_task = Task(title)
            new_task.id = max_id
            tasks.append(new_task)
            save_tasks(tasks)
            print("Задача добавлена.")
        # ... аналогично для остальных пунктов
        elif choice == '6':
            save_tasks(tasks)
            print("До свидания!")
            break
